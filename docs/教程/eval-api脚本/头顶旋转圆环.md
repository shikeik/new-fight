# 头顶旋转圆环

效果：在玩家 1 头顶生成一个黄色半透明圆环，并持续倾斜旋转。

## 创建（一次性调用）

```bash
eval-api <<'EOF'
(function(){
  const head = p1 && p1.head;
  if (!head) return 'no p1.head';

  // 移除旧的
  const old = head.getObjectByName('yellow-glow-ring');
  if (old) head.remove(old);
  modTick && modTick.remove && modTick.remove('ring-spin');

  // 创建圆环
  const ring = new THREE.Mesh(
    new THREE.TorusGeometry(0.6, 0.08, 16, 32),
    new THREE.MeshBasicMaterial({ color: 0xffff00, transparent: true, opacity: 0.8 })
  );
  ring.name = 'yellow-glow-ring';
  ring.position.set(0, 1.4, 0);
  ring.rotation.set(Math.PI / 3, 0, 0);
  head.add(ring);

  // 添加旋转任务
  if (typeof modTick !== 'undefined') {
    modTick.add('ring-spin', dt => {
      const r = p1.head.getObjectByName('yellow-glow-ring');
      if (r) { r.rotation.y += dt * 2.5; r.rotation.z += dt * 1.5; }
    });
  }

  return 'yellow glow ring created';
})()
EOF
```

## 移除

```bash
eval-api <<'EOF'
(function(){
  const head = p1 && p1.head;
  if (!head) return 'no p1.head';

  const old = head.getObjectByName('yellow-glow-ring');
  if (old) head.remove(old);
  if (typeof modTick !== 'undefined' && modTick.remove) {
    modTick.remove('ring-spin');
  }

  return 'yellow glow ring removed';
})()
EOF
```
